## 手順4: 共通 DOM ユーティリティを作成する

`src/utils/dom.js` を作成する。

```javascript
function clearChildren(element) {
  while (element.firstChild) {
    element.removeChild(element.firstChild);
  }
}

function createElement(tagName, attributes = {}, children = []) {
  const element = document.createElement(tagName);
  assignAttributes(element, attributes);
  element.append(...children);
  return element;
}

function createSvgElement(tagName, attributes = {}, children = []) {
  const element = document.createElementNS("http://www.w3.org/2000/svg", tagName);
  assignAttributes(element, attributes);
  element.append(...children);
  return element;
}

function assignAttributes(element, attributes) {
  for (const [key, value] of Object.entries(attributes)) {
    if (value === null || value === undefined) continue;
    element.setAttribute(key, String(value));
  }
}
```

## 手順5: データモデルを作成する

`src/models/diagramModel.js` を作成する。最初は、状態作成、クラス追加、クラス削除、関係追加、関係削除を実装する。

```javascript
let nextId = 1;

function createId(prefix) {
  nextId += 1;
  return `${prefix}_${nextId}`;
}

class DiagramModel {
  createInitialState() {
    return {
      version: 1,
      classes: [],
      relationships: [],
      viewport: { x: 0, y: 0, scale: 1 },
      selection: null
    };
  }

  addClass(state, input = {}) {
    const classNode = {
      id: createId("class"),
      name: input.name ?? this.createUniqueClassName(state),
      kind: input.kind ?? "class",
      stereotype: input.stereotype ?? "",
      visibility: input.visibility ?? "public",
      isAbstract: input.isAbstract ?? false,
      position: input.position ?? { x: 120, y: 100 },
      size: input.size ?? { width: 220, height: 150 },
      properties: input.properties ?? [],
      methods: input.methods ?? []
    };

    return {
      ...state,
      classes: [...state.classes, classNode],
      selection: { type: "class", id: classNode.id }
    };
  }

  updateClass(state, classId, patch) {
    return {
      ...state,
      classes: state.classes.map((classNode) => {
        if (classNode.id !== classId) return classNode;
        return { ...classNode, ...patch };
      })
    };
  }

  removeClass(state, classId) {
    return {
      ...state,
      classes: state.classes.filter((classNode) => classNode.id !== classId),
      relationships: state.relationships.filter((relationship) => (
        relationship.sourceClassId !== classId && relationship.targetClassId !== classId
      )),
      selection: null
    };
  }

  addRelationship(state, input) {
    const relationship = {
      id: createId("rel"),
      sourceClassId: input.sourceClassId,
      targetClassId: input.targetClassId,
      type: input.type ?? "association",
      label: input.label ?? "",
      sourceMultiplicity: input.sourceMultiplicity ?? "",
      targetMultiplicity: input.targetMultiplicity ?? ""
    };

    return {
      ...state,
      relationships: [...state.relationships, relationship],
      selection: { type: "relationship", id: relationship.id }
    };
  }

  removeRelationship(state, relationshipId) {
    return {
      ...state,
      relationships: state.relationships.filter((relationship) => relationship.id !== relationshipId),
      selection: null
    };
  }

  createUniqueClassName(state) {
    let index = state.classes.length + 1;
    let name = `NewClass${index}`;
    while (state.classes.some((classNode) => classNode.name === name)) {
      index += 1;
      name = `NewClass${index}`;
    }
    return name;
  }
}
```

```javascript
class StorageService {
  constructor(storageKey) {
    this.storageKey = storageKey;
  }

  load() {
    try {
      const raw = localStorage.getItem(this.storageKey);
      if (!raw) return null;
      return JSON.parse(raw);
    } catch {
      return null;
    }
  }

  save(state) {
    try {
      localStorage.setItem(this.storageKey, JSON.stringify(state));
      return true;
    } catch {
      return false;
    }
  }

  clear() {
    localStorage.removeItem(this.storageKey);
  }
}
```

```javascript

class SvgRenderer {
  constructor(svgElement) {
    this.svgElement = svgElement;
    this.handlers = {};
  }

  bindEvents(handlers) {
    this.handlers = handlers;
  }

  render(state) {
    clearChildren(this.svgElement);

    const defs = createSvgElement("defs", {}, [
      this.createMarker("arrow", "M 0 0 L 10 5 L 0 10 z"),
      this.createMarker("diamond", "M 0 5 L 6 0 L 12 5 L 6 10 z"),
      this.createMarker("triangle", "M 0 0 L 12 6 L 0 12 z", "white")
    ]);

    const viewportGroup = createSvgElement("g", {
      transform: `translate(${state.viewport.x} ${state.viewport.y}) scale(${state.viewport.scale})`
    });

    for (const relationship of state.relationships) {
      viewportGroup.append(this.renderRelationship(state, relationship));
    }

    for (const classNode of state.classes) {
      viewportGroup.append(this.renderClassNode(state, classNode));
    }

    this.svgElement.append(defs, viewportGroup);
  }

  renderClassNode(state, classNode) {
    const group = createSvgElement("g", {
      class: state.selection?.id === classNode.id ? "class-node selected" : "class-node",
      transform: `translate(${classNode.position.x} ${classNode.position.y})`,
      "data-class-id": classNode.id
    });

    group.append(
      createSvgElement("rect", {
        width: classNode.size.width,
        height: classNode.size.height,
        rx: 4
      }),
      createSvgElement("text", { x: 12, y: 24 }, [document.createTextNode(classNode.name)])
    );

    group.addEventListener("click", (event) => {
      event.stopPropagation();
      this.handlers.selectClass?.(classNode.id);
    });

    return group;
  }

  renderRelationship(state, relationship) {
    const source = state.classes.find((classNode) => classNode.id === relationship.sourceClassId);
    const target = state.classes.find((classNode) => classNode.id === relationship.targetClassId);
    if (!source || !target) return createSvgElement("g");

    const start = {
      x: source.position.x + source.size.width,
      y: source.position.y + source.size.height / 2
    };
    const end = {
      x: target.position.x,
      y: target.position.y + target.size.height / 2
    };
    const pathData = `M ${start.x} ${start.y} L ${end.x} ${end.y}`;

    const group = createSvgElement("g");
    group.append(
      createSvgElement("path", {
        class: `relationship-line ${relationship.type}`,
        d: pathData
      }),
      createSvgElement("path", {
        class: "relationship-hit",
        d: pathData,
        "data-relationship-id": relationship.id
      })
    );

    group.addEventListener("click", (event) => {
      event.stopPropagation();
      this.handlers.selectRelationship?.(relationship.id);
    });

    return group;
  }

  createMarker(id, pathData, fill = "#344054") {
    return createSvgElement("marker", {
      id,
      markerWidth: 12,
      markerHeight: 12,
      refX: 12,
      refY: 6,
      orient: "auto",
      markerUnits: "strokeWidth"
    }, [
      createSvgElement("path", {
        d: pathData,
        fill,
        stroke: "#344054"
      })
    ]);
  }
}
```

```javascript
class InspectorView {
  constructor(rootElement) {
    this.rootElement = rootElement;
    this.handlers = {};
  }

  bindEvents(handlers) {
    this.handlers = handlers;
  }

  render(state) {
    clearChildren(this.rootElement);
    const selection = state.selection;
    if (!selection) {
      this.rootElement.append(createElement("p", {}, [document.createTextNode("要素が選択されていません。")]));
      return;
    }

    if (selection.type === "class") {
      const classNode = state.classes.find((item) => item.id === selection.id);
      if (classNode) this.renderClassForm(classNode);
    }
  }

  renderClassForm(classNode) {
    const nameInput = createElement("input", {
      type: "text",
      value: classNode.name
    });

    nameInput.addEventListener("input", () => {
      this.handlers.updateClass?.(classNode.id, { name: nameInput.value });
    });

    this.rootElement.append(
      createElement("h2", {}, [document.createTextNode("クラス")]),
      createElement("label", {}, [document.createTextNode("名前"), nameInput])
    );
  }
}
```

## 手順9: Mermaid 変換サービスを作成する

`src/services/mermaidSerializer.js` を作成する。

```javascript
class MermaidSerializer {
  serialize(state) {
    const lines = ["classDiagram"];
    for (const classNode of state.classes) {
      lines.push(`  class ${safeName(classNode.name)} {`);
      for (const property of classNode.properties) {
        lines.push(`    ${visibilityMark(property.visibility)}${property.name} ${property.type}`);
      }
      for (const method of classNode.methods) {
        lines.push(`    ${visibilityMark(method.visibility)}${method.name}() ${method.returnType ?? ""}`.trimEnd());
      }
      lines.push("  }");
    }
    for (const relationship of state.relationships) {
      const source = state.classes.find((item) => item.id === relationship.sourceClassId);
      const target = state.classes.find((item) => item.id === relationship.targetClassId);
      if (!source || !target) continue;
      lines.push(`  ${safeName(source.name)} ${relationshipMark(relationship.type)} ${safeName(target.name)}`);
    }
    return lines.join("\n");
  }
}

function visibilityMark(visibility) {
  return { public: "+", protected: "#", private: "-", package: "~" }[visibility] ?? "+";
}

function relationshipMark(type) {
  return {
    inheritance: "<|--",
    implementation: "<|..",
    dependency: "..>",
    association: "-->",
    aggregation: "o--",
    composition: "*--"
  }[type] ?? "-->";
}

function safeName(name) {
  return String(name).trim().replace(/\s+/g, "_");
}
```

`src/services/mermaidParser.js` を作成する。

```javascript
class MermaidParser {
  parse(text) {
    const lines = text.split(/\r?\n/).map((line) => line.trim()).filter(Boolean);
    const classes = [];
    const relationships = [];

    for (const line of lines) {
      if (line === "classDiagram") continue;
      const classMatch = line.match(/^class\s+(.+?)(\s*\{)?$/);
      if (classMatch && !line.includes("--") && !line.includes("..")) {
        classes.push(createClass(classMatch[1]));
        continue;
      }

      const relationMatch = line.match(/^(.+?)\s+(<\|--|<\|..|\.\.>|-->|o--|\*--)\s+(.+?)(\s*:\s*(.+))?$/);
      if (relationMatch) {
        const sourceName = relationMatch[1].trim();
        const targetName = relationMatch[3].trim();
        const source = findOrCreateClass(classes, sourceName);
        const target = findOrCreateClass(classes, targetName);
        relationships.push({
          id: createId("rel"),
          sourceClassId: source.id,
          targetClassId: target.id,
          type: relationshipType(relationMatch[2]),
          label: relationMatch[5] ?? "",
          sourceMultiplicity: "",
          targetMultiplicity: ""
        });
      }
    }

    if (classes.length === 0 && relationships.length === 0) {
      throw new Error("Mermaidからクラス図情報を読み取れませんでした。");
    }

    return {
      version: 1,
      classes,
      relationships,
      viewport: { x: 0, y: 0, scale: 1 },
      selection: null
    };
  }
}
```

## 手順12: コントローラーと初期化を作成する

```javascript
class AppController {
  constructor(dependencies) {
    Object.assign(this, dependencies);
    this.state = this.model.createInitialState();
  }

  init() {
    this.state = this.storage.load() ?? this.model.createInitialState();
    this.bindToolbar();
    this.renderer.bindEvents({
      selectClass: (classId) => this.select("class", classId),
      selectRelationship: (relationshipId) => this.select("relationship", relationshipId)
    });
    this.inspector.bindEvents({
      updateClass: (classId, patch) => this.updateClass(classId, patch)
    });
    this.render();
  }

  bindToolbar() {
    document.querySelector("#add-class-button").addEventListener("click", () => this.addClass());
    document.querySelector("#delete-button").addEventListener("click", () => this.deleteSelected());
    document.querySelector("#export-button").addEventListener("click", () => {
      this.modal.showExport(this.serializer.serialize(this.state));
    });
    document.querySelector("#clear-button").addEventListener("click", () => {
      this.storage.clear();
      this.state = this.model.createInitialState();
      this.render();
    });
  }

  addClass() {
    this.state = this.model.addClass(this.state);
    this.render();
  }

  updateClass(classId, patch) {
    this.state = this.model.updateClass(this.state, classId, patch);
    this.render();
  }

  select(type, id) {
    this.state = { ...this.state, selection: { type, id } };
    this.render();
  }

  deleteSelected() {
    if (!this.state.selection) return;
    if (this.state.selection.type === "class") {
      this.state = this.model.removeClass(this.state, this.state.selection.id);
    }
    if (this.state.selection.type === "relationship") {
      this.state = this.model.removeRelationship(this.state, this.state.selection.id);
    }
    this.render();
  }

  render() {
    this.renderer.render(this.state);
    this.inspector.render(this.state);
    this.storage.save(this.state);
  }
}
```
